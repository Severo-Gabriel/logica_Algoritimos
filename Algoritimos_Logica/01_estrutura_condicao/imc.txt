algoritmo "calculo_imc"

var
   Ps, At, IMC: real

inicio

   Escreva("Indique seu peso: ")
   Leia(Ps)

   Escreva("Indique sua altura: ")
   Leia(At)

   IMC <- Ps / (At * At)

   Se (IMC > 22) entao
      Escreva("Você está acima da norma. Seu resultado: ", IMC)
   Senao
      Escreva("Você está abaixo da norma. Seu resultado: ", IMC)
   FimSe

fimalgoritmo
